import { Component, OnInit } from '@angular/core';

import { Subject } from 'rxjs/Subject';
import { Observable } from 'rxjs/Observable';

import { Noise } from './signals';

import * as lib from 'sigplot-ts';

import {
  PlotData,
  HighlightCommand,
  HighlightAction
} from './components';

/**
 * Example top-level application that implements two plots, a line and a raster,
 * and ties the data input from a noise signal generator to each.  In this example,
 * we've arbitrarily provided format, type, and units settings to show some of
 * the available control integration.
 */
@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.css']
})
export class AppComponent implements OnInit {
  title = 'SigPlot Angular Components Example App';

  /** Color map for the raster plot */
  rasterColorMap:     lib.Mc;

  /** Line plot's data observable */
  linePlotData$:      Observable<PlotData>;
  /** Line plot's highlight command observable */
  linePlotHighlight$: Observable<HighlightCommand>;
  /** Raster plot's data observable */
  rasterPlotData$:    Observable<PlotData>;

  /** Simple 'noise' signal generator */
  private signal:     Noise;

  /** Subject for the line plot's data */
  private _lpdSubject:  Subject<PlotData>;

  /** Subject for the line plot's highlight commands */
  private _lphSubject:  Subject<HighlightCommand>;

  /** The current highlight command */
  private _highlight:   HighlightCommand;

  /** Subject for the raster plot's data */
  private _rpdSubject: Subject<PlotData>;


  constructor() {
    // Pick the initial color map
    this.rasterColorMap = lib.Mc.Greyscale;

    // Setup the observable(s)
    this._lpdSubject = new Subject<PlotData>();
    this.linePlotData$ = this._lpdSubject.asObservable();
    this._lphSubject = new Subject<HighlightCommand>();
    this.linePlotHighlight$ = this._lphSubject.asObservable();

    this._rpdSubject = new Subject<PlotData>();
    this.rasterPlotData$ = this._rpdSubject.asObservable();

    // Setup signal generator
    this.signal = new Noise();
    this.signal.updateRateHz = 4;

    // Highlight's start/end are relative to the values being plotted, which is
    // naturally based on the delta between sample points.
    const delta: number = 1.0 / this.signal.updateRateHz / this.signal.frameSize;

    // Setup a highlight
    this._highlight = {
      action: HighlightAction.remove,
      highlight: {
        xstart: 100.0 * delta,
        xend:   400.0 * delta,
        color:  'red',
        id:     'my-highlight'
      }
    };

    // Connect the signal stream to the plot via 'packet'
    this.signal.start((b) => {
      // Line plot's data
      const lpd: PlotData = {
        buffer:   b,
        dataSize: lib.FormatSize.Scalar,
        dataType: lib.FormatType.Float32,
        xAxis: {
          start: 0.0,
          delta: delta,
          units: lib.Units.Time
        }
      };
      this._lpdSubject.next(lpd);

      // Falling raster version with y-axis data specs.
      const rpd: PlotData = {
        buffer:   b,
        dataSize: lib.FormatSize.Scalar,
        dataType: lib.FormatType.Float32,
        xAxis: {
          start: 0.0,
          delta: delta,
          units: lib.Units.Frequency
        },
        yAxis: {
          start: 0.0,
          units: lib.Units.Power
        }
      };
      this._rpdSubject.next(rpd);
    });
  }

  ngOnInit() {
    this.rasterColorMap = lib.Mc.Greyscale;
  }

  toggleHighlight() {
    if (this._highlight.action === HighlightAction.remove) {
      this._highlight.action = HighlightAction.add;
    } else {
      this._highlight.action = HighlightAction.remove;
    }
    console.log('Also pushing highlight');
    this._lphSubject.next(this._highlight);
  }

  changeRasterColorMap() {
    // This method of iterating over enumerations works here since this one is
    // defined positively from 0 without skipping.
    if (lib.Mc[this.rasterColorMap + 1] === undefined) {
      this.rasterColorMap = 0;
    } else {
      this.rasterColorMap += 1;
    }
    console.log(this.rasterColorMap + ' -> ' + lib.Mc[this.rasterColorMap])
  }
}
